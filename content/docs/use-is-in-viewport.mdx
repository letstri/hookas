---
title: useIsInViewport
description: React hook to detect if an element is visible in the viewport using Intersection Observer.
---

## Installation

```bash
npx shadcn@latest add https://hookas.letstri.dev/r/use-is-in-viewport.json
```

## Usage

```tsx
import { useIsInViewport } from './hooks/use-is-in-viewport'

function LazyImage() {
  const ref = useRef<HTMLImageElement>(null)
  const isVisible = useIsInViewport(ref, 'partial')

  return (
    <img
      ref={ref}
      src={isVisible ? 'actual-image.jpg' : 'placeholder.jpg'}
      alt="Lazy loaded image"
    />
  )
}
```
