---
title: useIsScrolled
description: React hook to detect if an element has been scrolled in a specific direction.
---

## Installation

```bash
npx shadcn@latest add https://hookas.letstri.dev/r/use-is-scrolled.json
```

## Usage

```tsx
import { useIsScrolled } from './hooks/use-is-scrolled'

function ScrollIndicator() {
  const ref = useRef<HTMLDivElement>(null)
  const isScrolled = useIsScrolled(ref, {
    direction: 'vertical',
    side: 'top',
    threshold: 10
  })

  return (
    <div>
      <div ref={ref} className="h-64 overflow-auto">
        {/* Scrollable content */}
      </div>
      {isScrolled && <div>Element has been scrolled!</div>}
    </div>
  )
}
```
