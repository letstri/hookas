## Usage

```tsx
import { useRef } from 'react'
import { useMediaControls } from './hooks/use-media-controls'

function App() {
  const videoRef = useRef<HTMLVideoElement>(null)
  const controls = useMediaControls(videoRef)

  return (
    <div>
      <video ref={videoRef} src="video.mp4" />

      <div>
        <button onClick={controls.toggle}>
          {controls.isPlaying ? 'Pause' : 'Play'}
        </button>
        <button onClick={controls.stop}>Stop</button>
        <button onClick={() => controls.toggleMute()}>
          {controls.isMuted ? 'Unmute' : 'Mute'}
        </button>

        <input
          type="range"
          min="0"
          max="1"
          step="0.1"
          value={controls.volume}
          onChange={(e) => controls.setVolume(Number.parseFloat(e.target.value))}
        />

        <input
          type="range"
          min="0"
          max={controls.duration}
          value={controls.currentTime}
          onChange={(e) => controls.setCurrentTime(Number.parseFloat(e.target.value))}
        />
      </div>
    </div>
  )
}
```
