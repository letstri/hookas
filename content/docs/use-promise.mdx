title: Use Promise

#### Installation

```bash
npx shadcn@latest add https://hookas.letstri.dev/r/use-promise.json
```

import { PromiseDemo } from '@/components/demo/use-promise'

#### Usage

```tsx
import { usePromise } from '@/hookas/use-promise'

function UserProfile() {
  // With default value
  const user = usePromise(() => fetchUser(), { name: 'Loading...' })

  // Without default (starts as undefined)
  const posts = usePromise(() => fetchUserPosts())

  return (
    <div>
      <h1>{user.name}</h1>
      {posts ? (
        <ul>
          {posts.map(post => <li key={post.id}>{post.title}</li>)}
        </ul>
      ) : (
        <p>Loading posts...</p>
      )}
    </div>
  )
}
```
