## Installation

```bash
npx shadcn@latest add https://hookas.letstri.dev/r/use-scroll-info.json
```

## Usage

```tsx
import { useScrollInfo } from './hooks/use-scroll-info'

function InfiniteScroll() {
  const ref = useRef<HTMLDivElement>(null)
  const { top, bottom, left, right } = useScrollInfo(ref)

  // Load more content when near bottom
  useEffect(() => {
    if (bottom < 100) {
      loadMoreContent()
    }
  }, [bottom])

  return (
    <div ref={ref} className="overflow-auto h-64">
      {/* Content */}
    </div>
  )
}
```
